CREATE TABLE Institution
(
    id serial primary key,
    title varchar(50)
);

CREATE TABLE Cafedra
(
    code varchar(4) primary key,
    title varchar(50),
    institutionId int,
    FOREIGN KEY (institutionId) REFERENCES Institution(id)
);

CREATE TABLE Specialty
(
    code varchar(8) primary key,
    title varchar(50),
    cafedraCode varchar(4),
    FOREIGN KEY (cafedraCode) REFERENCES Cafedra(code)
);

CREATE TABLE Course
(
    id serial primary key,
    title varchar(50),
    cafCode varchar(4),
    FOREIGN KEY (cafCode) REFERENCES Cafedra(code)
);

⦁	Создаем таблицу занятий

CREATE TABLE Lesson(
    id serial primary key,
    Title varchar(200),
    courseid int,
    FOREIGN KEY (courseid) REFERENCES Course(id)
);

⦁	Создаем таблицу с Группами
 CREATE TABLE Groups
(
code varchar(10) primary key,
cafedra varchar(4),
spec varchar(8),
EndYear int,
FOREIGN KEY (spec) REFERENCES Specialty(code) 
);

⦁	Создаем таблицу со студентами
create table STUDENTS
(
   code varchar(6) primary key,
   FULLNAME varchar(60),
   gcode varchar(10),
   FOREIGN KEY (gcode) REFERENCES groups(code)
);

⦁	Создаем таблицу с расписанием
CREATE TABLE Timetable
(
id serial primary key,
date timestamp, 
groupId varchar(10),
lessonId int, 
lessonNum int, 
FOREIGN KEY (groupId) REFERENCES Groups(code), 
FOREIGN KEY (lessonId) REFERENCES Lesson(id)
);

⦁	Создаем общую таблицу с посещениями
create table VISIT
(
id serial,
studentId varchar(6),
isVisited bool,
tt_id int,
date date,
FOREIGN KEY (studentId) REFERENCES Students(code),
FOREIGN KEY (tt_id) REFERENCES Timetable(id)
);
⦁	Создаем таблицу Группа-Студент
CREATE TABLE GroupStudent
(
	id serial primary key,
	code_st varchar(6),
	code_group varchar(10),
	FOREIGN KEY (code_st) REFERENCES Students(code),
	FOREIGN KEY (code_group) REFERENCES Groups(code)
);
